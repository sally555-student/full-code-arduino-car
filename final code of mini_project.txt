#include <ESP8266WiFi.h>  //for the use of esp to wifi

// SERVER 
ESP8266WebServer server(80); // the esp will create a server named "server" , working on port 80 default port for http, 
Servo myservo; 

// PINS  configurations
#define IN1 D1  //for motor A
#define IN2 D2
#define IN3 D3  // FOR MOTOR B
#define IN4 D4
#define PIR D5    
#define BUZZER D6  

// ULTRASONIC pins
#define TRIG D7     // triger the motion
#define ECHO D8     // send back the motion

int speedVal = 800;  
float distance = 0;  //the distance for the ultrasonic sensor
bool motionEnabled =false; //ffor the hasses l harake

void setup() {  
  Serial.begin(115200);  //for the serial monitor the baud rate

  // MOTOR PINS to tell the esp how each pin will behave 
// output: esp sends voltage, input : esp reads voltage
  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT);
  pinMode(IN4, OUTPUT);

  // configure the ULTRASONIC sensor
  pinMode(TRIG, OUTPUT);   // trig sends short pulse 
  pinMode(ECHO, INPUT);    // echo receives the reflected pulse

  // conf sensor motion 
  pinMode(PIR,INPUT);     //pir sensor sends high or low to the esp  
  pinMode(BUZZER,OUTPUT);   //esp controls when buzzer turns  on or off
  digitalWrite(BUZZER,LOW);  //turn off the buzzer initially

  // SERVO
  myservo.attach(D0);   // means the servo signal wire is connected to pin D0
  myservo.write(90);    

  // PWM CONFIG
  analogWriteRange(1023);  // range of 0 till 1023 speed dor pmw valkues

  // creat WIFI network 
  WiFi.softAP("SMART_DASH", "12345678"); // esp beacomes a wifi access point like a mini router, an end point will connects directly
      // to wwif name (SSID) ; smart-dash  and the password : 1234567

  // ROUTES   means if someone visit the URL , the fct will be executed
  // syntax : server.on("URL", functionname)
  //server is the web server we named . 
  server.on("/", landingPage);
  server.on("/dash", dashboard); // browser request http://192.268.4.1/dash ; esp executes dashborad();

  server.on("/move", handleMove);
  server.on("/speed", handleSpeed);
  server.on("/servo", handleServo);
  server.on("/distance", getDistance);
  server.on("/alert", alertBox);
  server.on("/toggleMotion",handleToggleMotion);
  server.on("/motionStatus",getMotionStatus);

  server.begin(); 

  stopMotor();    // INIT state
}

void loop() {
  server.handleClient();  //to check if client send a request  
  readDistance();  //so the distance is measured  by the esp 
  Serial.println(distance);
  
}

//  ULTRASONIC 
void readDistance() {
  digitalWrite(TRIG, LOW);  
  delayMicroseconds(2);
  digitalWrite(TRIG, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG, LOW);

  long duration = pulseIn(ECHO, HIGH, 30000);
  distance = duration * 0.034 / 2;
}

void getDistance() {
  server.send(200, "text/plain", String(distance));
}

void alertBox() {
  if (distance > 0 && distance < 7)
    server.send(200, "text/plain", "1");
  else
    server.send(200, "text/plain", "0");
}

//  HANDLERS 
void handleMove() {
  String d = server.arg("dir");  //get dir from the url , then call the corresponfing motor function

  if (d == "F") forward();
  else if (d == "B") backward();
  else if (d == "L") left();
  else if (d == "R") right();
  else stopMotor();

  server.send(200, "text/plain", "OK");
}

void handleSpeed() {
  speedVal = server.arg("val").toInt();  // reads slider value and stores in the variable

  server.send(200, "text/plain", "OK");
}

void handleServo() {  //to read the angle from the user commands and rotates the servo accordingly 
  myservo.write(server.arg("angle").toInt());
  server.send(200, "text/plain", "OK");
}

void handleToggleMotion() {  // toggle the motion and rthen send it to the webpage
  motionEnabled = !motionEnabled;
  server.send(200, "text/plain", motionEnabled ? "ON" : "OFF");
}

void getMotionStatus() {
  if (!motionEnabled) {
    digitalWrite(BUZZER, LOW);
    server.send(200, "text/plain", "OFF");
    return;
  }

  int pir = digitalRead(PIR);

  if (pir == HIGH) {
    digitalWrite(BUZZER, HIGH);
    server.send(200, "text/plain", "MOTION!");
  } else {
    digitalWrite(BUZZER, LOW);
    server.send(200, "text/plain", "NO MOTION");
  }
}

// MOTOR FUNCTIONS 
void forward() {  //1 for high and 0 for low  in this way the motors move forward  
  digitalWrite(IN1, 1); 
  digitalWrite(IN2, 0);
  digitalWrite(IN3, 1); 
  digitalWrite(IN4, 0);
}

void backward() {
  digitalWrite(IN1, 0); 
  digitalWrite(IN2, 1);
  digitalWrite(IN3, 0); 
  digitalWrite(IN4, 1);
}

void left() {
  digitalWrite(IN1, 0); 
  digitalWrite(IN2, 1);
  digitalWrite(IN3, 1); 
  digitalWrite(IN4, 0);
}

void right() {
  digitalWrite(IN1, 1); 
  digitalWrite(IN2, 0);
  digitalWrite(IN3, 0); 
  digitalWrite(IN4, 1);
}

void stopMotor() {
  digitalWrite(IN1, 0); 
  digitalWrite(IN2, 0);
  digitalWrite(IN3, 0); 
  digitalWrite(IN4, 0);
}
